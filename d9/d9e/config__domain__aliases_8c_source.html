53 /*! \brief Apply handler for domain_alias type */
54 static int domain_alias_apply(const struct ast_sorcery *sorcery, void *obj)
55 {
56  struct ast_sip_domain_alias *alias = obj;
57 
58  if (ast_strlen_zero(alias->domain)) {
59  /*
60  * What is the point of defining an alias and not saying
61  * what is being aliased?
62  */
63  ast_log(LOG_ERROR, "%s '%s' missing required domain being aliased.\n",
64  SIP_SORCERY_DOMAIN_ALIAS_TYPE, ast_sorcery_object_get_id(alias));
65  return -1;
66  }

71 int ast_sip_initialize_sorcery_domain_alias(void)
72 {
73  struct ast_sorcery *sorcery = ast_sip_get_sorcery();
74 
75  ast_sorcery_apply_default(sorcery, SIP_SORCERY_DOMAIN_ALIAS_TYPE, "config", "pjsip.conf,criteria=type=domain_alias");
76 
77  if (ast_sorcery_object_register(sorcery, SIP_SORCERY_DOMAIN_ALIAS_TYPE,
78  domain_alias_alloc, NULL, domain_alias_apply)) {
79  return -1;
80  }
81 
82  ast_sorcery_object_field_register(sorcery, SIP_SORCERY_DOMAIN_ALIAS_TYPE, "type", "",
83  OPT_NOOP_T, 0, 0);
84  ast_sorcery_object_field_register(sorcery, SIP_SORCERY_DOMAIN_ALIAS_TYPE, "domain",
85  "", OPT_STRINGFIELD_T, 0, STRFLDSET(struct ast_sip_domain_alias, domain));
86 
87  return 0;
