83 static int acf_version_exec(struct ast_channel *chan, const char *cmd,
84  char *parse, char *buffer, size_t buflen)
85 {
86  const char *response_char = ast_get_version();
87  AST_DECLARE_APP_ARGS(args,
88  AST_APP_ARG(info);
89  );
90 
91  AST_STANDARD_APP_ARGS(args, parse);
92 
93  if (!ast_strlen_zero(args.info) ) {
94  if (!strcasecmp(args.info,"ASTERISK_VERSION_NUM"))
95  response_char = ast_get_version_num();
96  else if (!strcasecmp(args.info,"BUILD_USER"))
97  response_char = BUILD_USER;
98  else if (!strcasecmp(args.info,"BUILD_HOSTNAME"))
99  response_char = BUILD_HOSTNAME;
100  else if (!strcasecmp(args.info,"BUILD_MACHINE"))
101  response_char = BUILD_MACHINE;
102  else if (!strcasecmp(args.info,"BUILD_KERNEL"))
103  response_char = BUILD_KERNEL;
104  else if (!strcasecmp(args.info,"BUILD_OS"))
105  response_char = BUILD_OS;
106  else if (!strcasecmp(args.info,"BUILD_DATE"))
107  response_char = BUILD_DATE;
108  }
109 
110  ast_debug(1, "VERSION returns %s result, given %s argument\n", response_char, args.info);
111 
112  ast_copy_string(buffer, response_char, buflen);
113 
114  return 0;
115 }
