137 /*! AO2 hash function for \ref message_subscription */
138 static int message_subscription_hash_cb(const void *obj, const int flags)
139 {
140  const struct message_subscription *sub;
141  const char *key;
142 
143  switch (flags & OBJ_SEARCH_MASK) {
144  case OBJ_SEARCH_KEY:
145  key = obj;
146  break;
147  case OBJ_SEARCH_OBJECT:
148  sub = obj;
149  key = sub->token;
150  break;
151  default:
152  /* Hash can only work on something with a full key. */
153  ast_assert(0);
154  return 0;
155  }
156  return ast_str_hash(key);

159 /*! AO2 comparison function for \ref message_subscription */
160 static int message_subscription_compare_cb(void *obj, void *arg, int flags)
161 {
162  const struct message_subscription *object_left = obj;
163  const struct message_subscription *object_right = arg;
164  const char *right_key = arg;
165  int cmp;
166 
167  switch (flags & OBJ_SEARCH_MASK) {
168  case OBJ_SEARCH_OBJECT:
169  right_key = object_right->token;
170  /* Fall through */
171  case OBJ_SEARCH_KEY:
172  cmp = strcmp(object_left->token, right_key);
173  break;
174  case OBJ_SEARCH_PARTIAL_KEY:
175  /*
176  * We could also use a partial key struct containing a length
177  * so strlen() does not get called for every comparison instead.
178  */
179  cmp = strncmp(object_left->token, right_key, strlen(right_key));
180  break;
181  default:
182  /*
183  * What arg points to is specific to this traversal callback
184  * and has no special meaning to astobj2.
185  */
186  cmp = 0;
187  break;
188  }
189  if (cmp) {
190  return 0;
191  }
192  /*
193  * At this point the traversal callback is identical to a sorted
194  * container.
195  */
196  return CMP_MATCH;
197 }

209 /*! \internal
210  * \brief Callback from the \c message API that determines if we can handle
211  * this message
212  */
213 static int has_destination_cb(const struct ast_msg *msg)
214 {
215  struct message_subscription *sub;
216  int i;
217  char buf[256];
218 
219  msg_to_endpoint(msg, buf, sizeof(buf));
220 
221  ast_rwlock_rdlock(&tech_subscriptions_lock);
222  for (i = 0; i < AST_VECTOR_SIZE(&tech_subscriptions); i++) {
223  sub = AST_VECTOR_GET(&tech_subscriptions, i);
224 
225  if (!sub) {
226  continue;
227  }
228 
229  if (!strcmp(sub->token, TECH_WILDCARD)
230  || !strncasecmp(sub->token, buf, strlen(sub->token))
231  || !strncasecmp(sub->token, buf, strlen(sub->token))) {
232  ast_rwlock_unlock(&tech_subscriptions_lock);
233  goto match;
234  }
235 
236  }
237  ast_rwlock_unlock(&tech_subscriptions_lock);
238 
239  sub = ao2_find(endpoint_subscriptions, buf, OBJ_SEARCH_KEY);
240  if (sub) {
241  ao2_ref(sub, -1);
242  goto match;
243  }
244 
245  ast_debug(1, "No subscription found for %s\n", buf);
246  return 0;
247 
248 match:
249  return 1;
250 }

252 static struct ast_json *msg_to_json(struct ast_msg *msg)
253 {
254  struct ast_json *json_obj;
255  struct ast_json *json_vars;
256  struct ast_msg_var_iterator *it_vars;
257  const char *name;
258  const char *value;
259 
260  it_vars = ast_msg_var_iterator_init(msg);
261  if (!it_vars) {
262  return NULL;
263  }
264 
265  json_vars = ast_json_object_create();
266  if (!json_vars) {
267  ast_msg_var_iterator_destroy(it_vars);
268  return NULL;
269  }
270 
271  while (ast_msg_var_iterator_next_received(msg, it_vars, &name, &value)) {
272  struct ast_json *json_val = ast_json_string_create(value);
273  if (!json_val || ast_json_object_set(json_vars, name, json_val)) {
274  ast_json_unref(json_vars);
275  ast_msg_var_iterator_destroy(it_vars);
276  return NULL;
277  }
278 
279  ast_msg_var_unref_current(it_vars);
280  }
281  ast_msg_var_iterator_destroy(it_vars);
282 
283  json_obj = ast_json_pack("{s: s, s: s, s: s, s: o}",
284  "from", ast_msg_get_from(msg),
285  "to", ast_msg_get_to(msg),
286  "body", ast_msg_get_body(msg),
287  "variables", json_vars);
288 
289  return json_obj;
290 }

306 static int handle_msg_cb(struct ast_msg *msg)
307 {
308  /* We have at most 3 subscriptions: TECH_WILDCARD, tech itself, and endpoint. */
309  struct message_subscription *matching_subscriptions[3];
310  struct message_subscription *sub;
311  int i, j;
312  int result;
313  char buf[256];
314  const char *endpoint_name;
315  struct ast_json *json_msg;
316 
317  msg_to_endpoint(msg, buf, sizeof(buf));
318  endpoint_name = buf;
319  json_msg = msg_to_json(msg);
320  if (!json_msg) {
321  return -1;
322  }
323  result = -1;
324 
325  /* Find subscriptions to TECH_WILDCARD and to the endpoint's technology. */
326  ast_rwlock_rdlock(&tech_subscriptions_lock);
327  for (i = 0, j = 0; i < AST_VECTOR_SIZE(&tech_subscriptions) && j < 2; i++) {
328  sub = AST_VECTOR_GET(&tech_subscriptions, i);
329 
330  if (!sub) {
331  continue;
332  }
333 
334  if (!strcmp(sub->token, TECH_WILDCARD)
335  || !strncasecmp(sub->token, buf, strlen(sub->token))) {
336  ao2_ref(sub, +1);
337  matching_subscriptions[j++] = sub;
338  }
339  }
340  ast_rwlock_unlock(&tech_subscriptions_lock);
341 
342  /* Find the subscription to this particular endpoint. */
343  sub = ao2_find(endpoint_subscriptions, buf, OBJ_SEARCH_KEY);
344  if (sub) {
345  matching_subscriptions[j++] = sub;
346  }
347 
348  /* Dispatch the message to all matching subscriptions. */
349  for (i = 0; i < j; i++) {
350  sub = matching_subscriptions[i];
351 
352  dispatch_message(sub, endpoint_name, json_msg);
353 
354  ao2_ref(sub, -1);
355  result = 0;
356  }
357 
358  ast_json_unref(json_msg);
359  return result;
360 }

394 static struct message_subscription *get_subscription(struct ast_endpoint *endpoint)
395 {
396  struct message_subscription *sub = NULL;
397 
398  if (endpoint && !ast_strlen_zero(ast_endpoint_get_resource(endpoint))) {
399  sub = ao2_find(endpoint_subscriptions, ast_endpoint_get_id(endpoint), OBJ_SEARCH_KEY);
400  } else {
401  int i;
402 
403  ast_rwlock_rdlock(&tech_subscriptions_lock);
404  for (i = 0; i < AST_VECTOR_SIZE(&tech_subscriptions); i++) {
405  sub = AST_VECTOR_GET(&tech_subscriptions, i);
406 
407  if (sub && !strcmp(sub->token, endpoint ? ast_endpoint_get_tech(endpoint) : TECH_WILDCARD)) {
408  ao2_bump(sub);
409  break;

411 
412  /* Need to reset the pointer at this line to prevent from using the wrong subscription due to
413  * the token check failing.
414  */
415  sub = NULL;
416  }
417  ast_rwlock_unlock(&tech_subscriptions_lock);
418  }
419 
420  return sub;

423 void messaging_app_unsubscribe_endpoint(const char *app_name, const char *endpoint_id)
424 {
425  RAII_VAR(struct message_subscription *, sub, NULL, ao2_cleanup);
426  RAII_VAR(struct ast_endpoint *, endpoint, NULL, ao2_cleanup);
427 
428  endpoint = ast_endpoint_find_by_id(endpoint_id);
429  sub = get_subscription(endpoint);
430  if (!sub) {
431  return;
432  }
433 
434  ao2_lock(sub);
435  if (!is_app_subscribed(sub, app_name)) {
436  ao2_unlock(sub);
437  return;
438  }
439 
440  AST_VECTOR_REMOVE_CMP_UNORDERED(&sub->applications, app_name, application_tuple_cmp, ao2_cleanup);
441  if (AST_VECTOR_SIZE(&sub->applications) == 0) {
442  if (endpoint && !ast_strlen_zero(ast_endpoint_get_resource(endpoint))) {
443  ao2_unlink(endpoint_subscriptions, sub);
444  } else {
445  ast_rwlock_wrlock(&tech_subscriptions_lock);
446  AST_VECTOR_REMOVE_CMP_UNORDERED(&tech_subscriptions, endpoint ? ast_endpoint_get_id(endpoint) : TECH_WILDCARD,
447  messaging_subscription_cmp, AST_VECTOR_ELEM_CLEANUP_NOOP);
448  ast_rwlock_unlock(&tech_subscriptions_lock);
449  ao2_ref(sub, -1); /* Release the reference held by tech_subscriptions */
450  }
451  }
452  ao2_unlock(sub);
453 
454  ast_debug(3, "App '%s' unsubscribed to messages from endpoint '%s'\n", app_name, endpoint ? ast_endpoint_get_id(endpoint) : "-- ALL --");
455  ast_test_suite_event_notify("StasisMessagingSubscription", "SubState: Unsubscribed\r\nAppName: %s\r\nToken: %s\r\n",
456  app_name, endpoint ? ast_endpoint_get_id(endpoint) : "ALL");
457 }

459 static struct message_subscription *get_or_create_subscription(struct ast_endpoint *endpoint)
460 {
461  struct message_subscription *sub = get_subscription(endpoint);
462 
463  if (sub) {
464  return sub;
465  }
466 
467  sub = message_subscription_alloc(endpoint ? ast_endpoint_get_id(endpoint) : TECH_WILDCARD);
468  if (!sub) {
469  return NULL;
470  }
471 
472  /* Either endpoint_subscriptions or tech_subscriptions will hold a reference to
473  * the subscription. This reference is released to allow the subscription to
474  * eventually destruct when there are no longer any applications receiving
475  * events from the subscription.
476  */
477  if (endpoint && !ast_strlen_zero(ast_endpoint_get_resource(endpoint))) {
478  ao2_link(endpoint_subscriptions, sub);
479  } else {
480  ast_rwlock_wrlock(&tech_subscriptions_lock);
481  ao2_ref(sub, +1);
482  if (AST_VECTOR_APPEND(&tech_subscriptions, sub)) {
483  /* Release the refs that were for the vector and the allocation. */
484  ao2_ref(sub, -2);
485  sub = NULL;
486  }
487  ast_rwlock_unlock(&tech_subscriptions_lock);
488  }
489 
490  return sub;
491 }

493 int messaging_app_subscribe_endpoint(const char *app_name, struct ast_endpoint *endpoint, message_received_cb callback, void *pvt)
494 {
495  RAII_VAR(struct message_subscription *, sub, NULL, ao2_cleanup);
496  struct application_tuple *tuple;
497 
498  sub = get_or_create_subscription(endpoint);
499  if (!sub) {
500  return -1;
501  }
502 
503  ao2_lock(sub);
504  if (is_app_subscribed(sub, app_name)) {
505  ao2_unlock(sub);
506  return 0;
507  }
508 
509  tuple = application_tuple_alloc(app_name, callback, pvt);
510  if (!tuple) {
511  ao2_unlock(sub);
512  return -1;
513  }
514  if (AST_VECTOR_APPEND(&sub->applications, tuple)) {
515  ao2_ref(tuple, -1);
516  ao2_unlock(sub);
517  return -1;
518  }
519  ao2_unlock(sub);
520 
521  ast_debug(3, "App '%s' subscribed to messages from endpoint '%s'\n", app_name, endpoint ? ast_endpoint_get_id(endpoint) : "-- ALL --");

523  app_name, endpoint ? ast_endpoint_get_id(endpoint) : "ALL");
524 
525  return 0;
526 }
