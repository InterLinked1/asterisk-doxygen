271  ast_test_status_update(test, "Failed to get value of %zu bytes\n", ast_str_strlen(s));
272  res = AST_TEST_FAIL;

274  ast_test_status_update(test, "Failed to match value of %zu bytes\n", ast_str_strlen(s));
