60 extern DBM *
61 dbm_open(file, flags, mode)
62  const char *file;
63  int flags, mode;
64 {
65  DBM *db;
66  HASHINFO info;
67  const size_t len = strlen(file) + sizeof (DBM_SUFFIX);
68 #ifdef __GNUC__
69  char path[len];
70 #else
71  char *path = malloc(len);
72  if (path == NULL)
73  return NULL;
74 #endif
75 
76  info.bsize = 4096;
77  info.ffactor = 40;
78  info.nelem = 1;
79  info.cachesize = 0;
80  info.hash = NULL;
81  info.lorder = 0;
82  snprintf(path, len, "%s%s", file, DBM_SUFFIX);
83  db = (DBM *)__hash_open(path, flags, mode, &info, 0);
84 #ifndef __GNUC__
85  free(path);
86 #endif
87  return db;
88 }
